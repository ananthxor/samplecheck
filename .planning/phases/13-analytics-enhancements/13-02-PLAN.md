```typescript
export interface HourlyDataPoint {
  hour_of_day: number  // 0-23
  impressions: number
  clicks: number
}

export interface CreativePieSlice {
  name: string        // creative name or truncated ID
  creativeId: string
  value: number       // impressions_served
  share: number       // 0-100, one decimal place
}

export interface DeviceBreakdownPoint {
  device_type: string  // 'desktop' | 'mobile' | 'tablet' | 'unknown'
  impressions: number
}

/**
 * Aggregate DailyMetricRow[] into CreativePieSlice[] by creative.
 * Caps at top 8 creatives; groups remainder as "Other".
 * Used by CreativePieChart (ANLYT-10).
 */
export function aggregateByCreative(
  rows: DailyMetricRow[],
  creativeNames: Record<string, string>
): CreativePieSlice[] {
  const byCreative = new Map<string, number>()
  for (const row of rows) {
    const id = row.creative_id ?? 'unknown'
    byCreative.set(id, (byCreative.get(id) ?? 0) + row.impressions_served)
  }
  const total = Array.from(byCreative.values()).reduce((a, b) => a + b, 0)
  const sorted = Array.from(byCreative.entries())
    .map(([creativeId, value]) => ({
      name: creativeNames[creativeId] ?? creativeId.slice(0, 8),
      creativeId,
      value,
      share: total > 0 ? Number(((value / total) * 100).toFixed(1)) : 0,
    }))
    .sort((a, b) => b.value - a.value)

  if (sorted.length <= 8) return sorted

  const top8 = sorted.slice(0, 8)
  const otherValue = sorted.slice(8).reduce((sum, s) => sum + s.value, 0)
  const otherShare = total > 0 ? Number(((otherValue / total) * 100).toFixed(1)) : 0
  return [...top8, { name: 'Other', creativeId: 'other', value: otherValue, share: otherShare }]
}
```

```typescript
export async function fetchHourlyMetrics(
  advertiserId: string,
  date: string  // 'YYYY-MM-DD'
): Promise<HourlyDataPoint[]> {
  const { data, error } = await supabase.rpc('fetch_hourly_metrics', {
    p_advertiser_id: advertiserId,
    p_date: date,
  })
  if (error) throw new Error(error.message)
  return data as HourlyDataPoint[]
}

export async function fetchLifetimeMetrics(
  advertiserId: string
): Promise<DailyMetricRow[]> {
  // PERFORMANCE NOTE: Unbounded query on daily_metrics.
  // For v1, row count is small (new product). Revisit if latency > 2s post-launch.
  const { data, error } = await supabase
    .from('daily_metrics')
    .select(
      'metric_date, creative_id, campaign_id, impressions_served, impressions_viewable, clicks, engagements, video_plays, video_completes, total_dwell_time_ms'
    )
    .eq('advertiser_id', advertiserId)
    .limit(100_000)  // safety cap per research recommendation
  if (error) throw new Error(error.message)
  return data as DailyMetricRow[]
}

export async function fetchDeviceBreakdown(
  advertiserId: string,
  startDate: string,
  endDate: string
): Promise<DeviceBreakdownPoint[]> {
  const { data, error } = await supabase.rpc('fetch_device_breakdown', {
    p_advertiser_id: advertiserId,
    p_start_date: startDate,
    p_end_date: endDate,
  })
  if (error) throw new Error(error.message)
  return data as DeviceBreakdownPoint[]
}
```

**use-analytics.ts — append three new hooks (keep all existing hooks):**

```typescript
export function useHourlyMetrics(date: string | null) {
  const { profile } = useAuth()
  return useQuery({
    queryKey: ['analytics-hourly', profile?.advertiser_id, date],
    queryFn: () => fetchHourlyMetrics(profile!.advertiser_id!, date!),
    enabled: !!profile?.advertiser_id && !!date,
    staleTime: 5 * 60 * 1000,
  })
}

export function useLifetimeMetrics() {
  const { profile } = useAuth()
  return useQuery({
    queryKey: ['analytics-lifetime', profile?.advertiser_id],
    queryFn: () => fetchLifetimeMetrics(profile!.advertiser_id!),
    enabled: !!profile?.advertiser_id,
    staleTime: 10 * 60 * 1000,
  })
}

export function useDeviceBreakdown(startDate: string, endDate: string) {
  const { profile } = useAuth()
  return useQuery({
    queryKey: ['analytics-device', profile?.advertiser_id, startDate, endDate],
    queryFn: () => fetchDeviceBreakdown(profile!.advertiser_id!, startDate, endDate),
    enabled: !!profile?.advertiser_id,
    staleTime: 5 * 60 * 1000,
  })
}
```
